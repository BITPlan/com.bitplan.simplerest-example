25  public class TestJaxB {
26    boolean debug=true;
27    boolean moxy=true;
28  
29    @Test
30    public void testVersion() throws JAXBException {
31      JaxbFactory<User> userFactory = new JaxbFactory<User>(User.class);
32      // String jaxbContext = JAXBContext.newInstance(User.class)
33      String jaxbContextType=userFactory.getJAXBContext().getClass().getName();
34      if (debug)
35        System.out.println(jaxbContextType);
36      // Oracle/Sun default:
37      String expected = "com.sun.xml.bind.v2.runtime.JAXBContextImpl";
38      if (moxy)
39        expected = "org.eclipse.persistence.jaxb.JAXBContext"; // jersey-moxy
40      assertEquals("Expecting configured JaxB implementation", expected,
41          jaxbContextType);
42    }
43    
44    @Test
45    public void testJaxbFactory() throws Exception {
46      JaxbFactory<User> userFactory = new JaxbFactory<User>(User.class);
47      User user=new User();
48      user.name="Doe";
49      user.firstname="John";
50      String xml=userFactory.asXML(user);
51      if (debug)
52        System.out.println("'"+xml+"'");
53    }
