9   package com.bitplan.simplerest.example;
10  
11  import com.bitplan.rest.RestServerImpl;
12  
13  /**
14   * an Example Server
15   * @author wf
16   *
17   */
18  public class ExampleServer extends RestServerImpl { 
19    
20    /**
21     * construct Example Server
22     * setting defaults
23     * @throws Exception 
24     */
25    public ExampleServer() throws Exception {
26      // listen to whole network
27      settings.setHost("0.0.0.0");
28      // listen on given port
29      settings.setPort(8111);
30      // add the default path
31      settings.setContextPath("/example");
32      // add a static handler
33      settings.addClassPathHandler("/", "/static/");
34      // setup resources from the given packages
35      String packages="com.bitplan.simplerest.example.resources;";
36      settings.setPackages(packages);
37     }
38    
39    /**
40     * start Server
41     * 
42     * @param args
43     * @throws Exception
44     */
45     public static void main(String[] args) throws Exception {
46       ExampleServer rs=new ExampleServer();
47       rs.settings.parseArguments(args);
48       rs.startWebServer();
49     } // main
50  
51  }
